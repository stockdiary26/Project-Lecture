```sql
-- Supabase SQL Editor에서 실행

-- 반
CREATE TABLE classes (
    id BIGSERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 학생
CREATE TABLE students (
    id BIGSERIAL PRIMARY KEY,
    class_id BIGINT REFERENCES classes(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    school_name TEXT NOT NULL DEFAULT '',
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 영역 카테고리
CREATE TABLE categories (
    id BIGSERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    sort_order INTEGER NOT NULL DEFAULT 0,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 시험
CREATE TABLE exams (
    id BIGSERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    exam_type TEXT NOT NULL CHECK (exam_type IN ('중간', '기말', '모의고사')),
    exam_date DATE NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 점수
CREATE TABLE scores (
    id BIGSERIAL PRIMARY KEY,
    student_id BIGINT REFERENCES students(id) ON DELETE CASCADE,
    exam_id BIGINT REFERENCES exams(id) ON DELETE CASCADE,
    total_score INTEGER NOT NULL CHECK (total_score >= 0 AND total_score <= 100),
    grade INTEGER CHECK (grade >= 1 AND grade <= 9),
    created_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(student_id, exam_id)
);

-- 영역별 점수
CREATE TABLE category_scores (
    id BIGSERIAL PRIMARY KEY,
    score_id BIGINT REFERENCES scores(id) ON DELETE CASCADE,
    category_id BIGINT REFERENCES categories(id) ON DELETE CASCADE,
    score INTEGER NOT NULL CHECK (score >= 0),
    UNIQUE(score_id, category_id)
);

-- 상담문
CREATE TABLE consultations (
    id BIGSERIAL PRIMARY KEY,
    student_id BIGINT REFERENCES students(id) ON DELETE CASCADE,
    exam_id BIGINT REFERENCES exams(id) ON DELETE CASCADE,
    ai_draft TEXT NOT NULL DEFAULT '',
    final_text TEXT NOT NULL DEFAULT '',
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 메모
CREATE TABLE memos (
    id BIGSERIAL PRIMARY KEY,
    student_id BIGINT REFERENCES students(id) ON DELETE CASCADE,
    content TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 기본 카테고리 삽입
INSERT INTO categories (name, sort_order) VALUES
    ('문법', 1),
    ('독해', 2),
    ('문학', 3);
```

```python
import streamlit as st
from supabase import create_client, Client


def get_client() -> Client:
    """Supabase 클라이언트를 반환한다. 세션당 1회만 생성."""
    if "supabase" not in st.session_state:
        st.session_state.supabase = create_client(
            st.secrets["SUPABASE_URL"],
            st.secrets["SUPABASE_KEY"],
        )
    return st.session_state.supabase


# ── Classes ──────────────────────────────────────────

def get_classes() -> list[dict]:
    client = get_client()
    res = client.table("classes").select("*").order("name").execute()
    return res.data


def create_class(name: str) -> dict:
    client = get_client()
    res = client.table("classes").insert({"name": name}).execute()
    return res.data[0]


def update_class(class_id: int, name: str) -> dict:
    client = get_client()
    res = client.table("classes").update({"name": name}).eq("id", class_id).execute()
    return res.data[0]


def delete_class(class_id: int) -> None:
    client = get_client()
    client.table("classes").delete().eq("id", class_id).execute()


# ── Students ─────���───────────────────────────────────

def get_students(class_id: int | None = None) -> list[dict]:
    client = get_client()
    query = client.table("students").select("*, classes(name)")
    if class_id:
        query = query.eq("class_id", class_id)
    res = query.order("name").execute()
    return res.data


def get_students_by_school() -> list[dict]:
    client = get_client()
    res = (
        client.table("students")
        .select("*, classes(name)")
        .order("school_name")
        .order("name")
        .execute()
    )
    return res.data


def create_student(class_id: int, name: str, school_name: str) -> dict:
    client = get_client()
    res = client.table("students").insert({
        "class_id": class_id,
        "name": name,
        "school_name": school_name,
    }).execute()
    return res.data[0]


def update_student(student_id: int, **fields) -> dict:
    client = get_client()
    res = client.table("students").update(fields).eq("id", student_id).execute()
    return res.data[0]


def delete_student(student_id: int) -> None:
    client = get_client()
    client.table("students").delete().eq("id", student_id).execute()


def bulk_create_students(students: list[dict]) -> list[dict]:
    client = get_client()
    res = client.table("students").insert(students).execute()
    return res.data


# ── Categories ───────────────────────────────────────

def get_categories() -> list[dict]:
    client = get_client()
    res = client.table("categories").select("*").order("sort_order").execute()
    return res.data


def create_category(name: str, sort_order: int) -> dict:
    client = get_client()
    res = client.table("categories").insert({
        "name": name,
        "sort_order": sort_order,
    }).execute()
    return res.data[0]


def update_category(category_id: int, **fields) -> dict:
    client = get_client()
    res = client.table("categories").update(fields).eq("id", category_id).execute()
    return res.data[0]


def delete_category(category_id: int) -> None:
    client = get_client()
    client.table("categories").delete().eq("id", category_id).execute()


# ── Exams ────────────────────────────────────────────

def get_exams() -> list[dict]:
    client = get_client()
    res = client.table("exams").select("*").order("exam_date", desc=True).execute()
    return res.data


def create_exam(name: str, exam_type: str, exam_date: str) -> dict:
    client = get_client()
    res = client.table("exams").insert({
        "name": name,
        "exam_type": exam_type,
        "exam_date": exam_date,
    }).execute()
    return res.data[0]


def delete_exam(exam_id: int) -> None:
    client = get_client()
    client.table("exams").delete().eq("id", exam_id).execute()


# ── Scores ────────���──────────────────────────���───────

def get_scores_by_exam(exam_id: int) -> list[dict]:
    client = get_client()
    res = (
        client.table("scores")
        .select("*, students(name, school_name, class_id)")
        .eq("exam_id", exam_id)
        .execute()
    )
    return res.data


def get_scores_by_student(student_id: int) -> list[dict]:
    client = get_client()
    res = (
        client.table("scores")
        .select("*, exams(name, exam_type, exam_date)")
        .eq("student_id", student_id)
        .order("created_at")
        .execute()
    )
    return res.data


def upsert_score(student_id: int, exam_id: int, total_score: int,
                 grade: int | None = None) -> dict:
    client = get_client()
    data = {
        "student_id": student_id,
        "exam_id": exam_id,
        "total_score": total_score,
    }
    if grade is not None:
        data["grade"] = grade
    res = client.table("scores").upsert(
        data, on_conflict="student_id,exam_id"
    ).execute()
    return res.data[0]


# ── Category Scores ──────��───────────────────────────

def get_category_scores(score_id: int) -> list[dict]:
    client = get_client()
    res = (
        client.table("category_scores")
        .select("*, categories(name)")
        .eq("score_id", score_id)
        .order("categories(sort_order)")
        .execute()
    )
    return res.data


def get_category_scores_by_student(student_id: int) -> list[dict]:
    client = get_client()
    res = (
        client.table("category_scores")
        .select("*, scores(exam_id, exams(name, exam_date)), categories(name)")
        .eq("scores.student_id", student_id)
        .execute()
    )
    return res.data


def upsert_category_score(score_id: int, category_id: int, score: int) -> dict:
    client = get_client()
    res = client.table("category_scores").upsert(
        {"score_id": score_id, "category_id": category_id, "score": score},
        on_conflict="score_id,category_id",
    ).execute()
    return res.data[0]


# ── Consultations ────────────────────────────────────

def get_consultations_by_student(student_id: int) -> list[dict]:
    client = get_client()
    res = (
        client.table("consultations")
        .select("*, exams(name, exam_date)")
        .eq("student_id", student_id)
        .order("created_at", desc=True)
        .execute()
    )
    return res.data


def create_consultation(student_id: int, exam_id: int,
                        ai_draft: str, final_text: str = "") -> dict:
    client = get_client()
    res = client.table("consultations").insert({
        "student_id": student_id,
        "exam_id": exam_id,
        "ai_draft": ai_draft,
        "final_text": final_text or ai_draft,
    }).execute()
    return res.data[0]


def update_consultation(consultation_id: int, final_text: str) -> dict:
    client = get_client()
    res = (
        client.table("consultations")
        .update({"final_text": final_text})
        .eq("id", consultation_id)
        .execute()
    )
    return res.data[0]


# ── Memos ───────��────────────────────────────────────

def get_memos_by_student(student_id: int) -> list[dict]:
    client = get_client()
    res = (
        client.table("memos")
        .select("*")
        .eq("student_id", student_id)
        .order("created_at", desc=True)
        .execute()
    )

def create_memo(student_id: int, content: str) -> dict:

    client = get_client()
```

```python
import streamlit as st


def check_password() -> bool:
    """비밀번호 인증. 통과하면 True 반환."""
    if st.session_state.get("authenticated"):
        return True

    st.title("🔐 로그인")
    password = st.text_input("비밀번호를 입력하세요", type="password")

    if st.button("로그인"):
        if password == st.secrets["APP_PASSWORD"]:
            st.session_state.authenticated = True
            st.rerun()
        else:
            st.error("비밀번호가 올���르지 않습니다.")

    return False
```

```python
import streamlit as st
from lib.auth import check_password
from lib import db

st.set_page_config(
    page_title="국어 성적 관리",
    page_icon="📚",
    layout="wide",
)

if not check_password():
    st.stop()

students = db.get_students()
exams = db.get_exams()

with col1:
    st.metric("반", f"{len(classes)}개")
with col2:
    st.metric("학생", f"{len(students)}명")
with col3:
    st.metric("시험", f"{len(exams)}회")

# 최근 시험 요약
if exams:
    latest_exam = exams[0]
    st.subheader(f"📝 최근 시험: {latest_exam['name']}")
    st.write(f"유형: {latest_exam['exam_type']} | 날짜: {latest_exam['exam_date']}")

    scores = db.get_scores_by_exam(latest_exam["id"])
    if scores:
        avg = sum(s["total_score"] for s in scores) / len(scores)
        st.metric("평균 점수", f"{avg:.1f}점")

        consultations = []
        for s in scores:
            cons = db.get_consultations_by_student(s["student_id"])
            exam_cons = [c for c in cons if c.get("exam_id") == latest_exam["id"]]
            if exam_cons:
                consultations.append(s["student_id"])

        remaining = len(scores) - len(consultations)
        if remaining > 0:
            st.warning(f"💬 상담문 미작성 학생: {remaining}명")
        else:
            st.success("✅ 모든 학생 상담문 작성 완료")
    else:
        st.info("아직 점수가 입력되지 않았습니다.")

    # 온보딩 안내
    st.info("👋 시작하려면 사이드바에서 **학생 관리** → **성적 입력** 순으로 진행하세요.")
```

```python
def calc_score_change(scores: list[dict]) -> dict:
    """시험 점수 이력에서 최근 변화를 계산한다."""
    if not scores:
        return {"current": 0, "previous": None, "change": 0, "direction": "same"}

    sorted_scores = sorted(scores, key=lambda s: s["exams"]["exam_date"])
    current = sorted_scores[-1]["total_score"]

    if len(sorted_scores) < 2:
        return {"current": current, "previous": None, "change": 0, "direction": "same"}

    previous = sorted_scores[-2]["total_score"]
    change = current - previous

    if change > 0:
        direction = "up"
    elif change < 0:
        direction = "down"
    else:
        direction = "same"

    return {
        "current": current,
        "previous": previous,
        "change": change,
        "direction": direction,
    }


def calc_class_average(scores: list[dict]) -> float:
    """점수 목록의 평균을 계산한다. 소수점 1자리."""
    if not scores:
        return 0.0
    total = sum(s["total_score"] for s in scores)
    return round(total / len(scores), 1)


def rank_in_group(student_id: int, scores: list[dict]) -> dict:
    """학생의 그룹 내 등수를 계산한다. 내림차순."""
    sorted_scores = sorted(scores, key=lambda s: s["total_score"], reverse=True)
    for i, s in enumerate(sorted_scores):
        if s["student_id"] == student_id:
            return {"rank": i + 1, "total": len(sorted_scores)}
    return {"rank": 0, "total": len(sorted_scores)}


def calc_category_changes(
    current: list[dict], previous: list[dict]
) -> dict[str, dict]:
    """영역별 점수 변화를 계산한다."""
    prev_map = {c["category_name"]: c["score"] for c in previous}
    result = {}
    for cat in current:
        name = cat["category_name"]
        cur_score = cat["score"]
        prev_score = prev_map.get(name)
        result[name] = {
            "current": cur_score,
            "previous": prev_score,
            "change": cur_score - prev_score if prev_score is not None else 0,
        }
    return result


def find_strengths_weaknesses(categories: list[dict]) -> dict:
    """영역별 점수에서 강점과 약점을 찾는다."""
    if not categories:
        return {"strengths": [], "weaknesses": []}

    sorted_cats = sorted(categories, key=lambda c: c["score"], reverse=True)
    avg = sum(c["score"] for c in sorted_cats) / len(sorted_cats)

    strengths = [c["category_name"] for c in sorted_cats if c["score"] >= avg]
    weaknesses = [c["category_name"] for c in sorted_cats if c["score"] < avg]

    return {"strengths": strengths[:2], "weaknesses": weaknesses[:2]}
```

```python
st.subheader("📊 영역별 분석")

latest_score_id = sorted_scores[-1]["id"]
cat_scores = db.get_category_scores(latest_score_id)

if cat_scores:
    current_cats = [{"category_name": c["categories"]["name"], "score": c["score"]} for c in cat_scores]

    # 이전 시험 영역별 점수
    prev_cats = []
    if len(sorted_scores) >= 2:
        prev_score_id = sorted_scores[-2]["id"]
        prev_cat_scores = db.get_category_scores(prev_score_id)
        prev_cats = [{"category_name": c["categories"]["name"], "score": c["score"]} for c in prev_cat_scores]

    changes = calc_category_changes(current_cats, prev_cats)
    sw = find_strengths_weaknesses(current_cats)

    # 영역별 변화 표
    for name, data in changes.items():
        delta = f"{data['change']:+d}" if data["previous"] is not None else "첫 시험"
        st.metric(name, f"{data['current']}점", delta)

    if sw["strengths"]:
        st.success(f"💪 강점: {', '.join(sw['strengths'])}")
    if sw["weaknesses"]:
        st.warning(f"📝 보완: {', '.join(sw['weaknesses'])}")
else:
    st.info("영역별 점수가 입력되지 않았습니다.")
```

## Task 9: AI 상담문 생성 모듈

```python
import streamlit as st
import anthropic
from lib import db
from lib.analysis import (
    calc_score_change,
    calc_class_average,
    calc_category_changes,
    find_strengths_weaknesses,
    rank_in_group,
)


def get_anthropic_client() -> anthropic.Anthropic:
    """Anthropic 클라이언트를 반환한다."""
    return anthropic.Anthropic(api_key=st.secrets["ANTHROPIC_API_KEY"])


def build_student_context(student_id: int, exam_id: int) -> str:
    """학생의 성적 데이터를 텍스트 컨텍스트로 구성한다."""
    students = db.get_students()
    student = next((s for s in students if s["id"] == student_id), None)
    if not student:
        return ""

    scores = db.get_scores_by_student(student_id)
    if not scores:
        return ""

    # 점수 변화
    change = calc_score_change(scores)

    # 반 평균 + 순위
    exam_scores = db.get_scores_by_exam(exam_id)
    avg = calc_class_average(exam_scores)
    ranking = rank_in_group(student_id, exam_scores)

    # 영역별 분석
    sorted_scores = sorted(scores, key=lambda s: s["exams"]["exam_date"])
    latest_score = next((s for s in sorted_scores if s["exam_id"] == exam_id), sorted_scores[-1])
    cat_scores = db.get_category_scores(latest_score["id"])

    category_text = ""
    if cat_scores:
        current_cats = [{"category_name": c["categories"]["name"], "score": c["score"]} for c in cat_scores]

        prev_cats = []
        prev_score = None
        for s in sorted_scores:
            if s["id"] != latest_score["id"]:
                prev_score = s
        if prev_score:
            prev_cat_scores = db.get_category_scores(prev_score["id"])
            prev_cats = [{"category_name": c["categories"]["name"], "score": c["score"]} for c in prev_cat_scores]

        changes = calc_category_changes(current_cats, prev_cats)
        sw = find_strengths_weaknesses(current_cats)

        category_lines = []
        for name, data in changes.items():
            if data["previous"] is not None:
                category_lines.append(f"  - {name}: {data['current']}점 (이전 대비 {data['change']:+d})")
            else:
                category_lines.append(f"  - {name}: {data['current']}점")
        category_text = "\n".join(category_lines)

        if sw["strengths"]:
            category_text += f"\n  강점 영역: {', '.join(sw['strengths'])}"
        if sw["weaknesses"]:
            category_text += f"\n  보완 영역: {', '.join(sw['weaknesses'])}"

    # 이전 상담문
    prev_consultations = db.get_consultations_by_student(student_id)
    prev_text = ""
    if prev_consultations:
        latest_cons = prev_consultations[0]
        prev_text = f"\n이전 상담문:\n{latest_cons['final_text'][:500]}"

    # 메모
    memos = db.get_memos_by_student(student_id)
    memo_text = ""
    if memos:
        recent_memos = memos[:3]
        memo_lines = [f"  - {m['content']}" for m in recent_memos]
        memo_text = f"\n최근 메모:\n" + "\n".join(memo_lines)

    # 등급 정보
    grade_text = ""
    if latest_score.get("grade"):
        grade_text = f"\n내신 ��급: {latest_score['grade']}등급"

    context = f"""학생 정보:
- 이름: {student['name']}
- 학교: {student['school_name']}
{grade_text}

이번 시험 성적:
- 총점: {change['current']}점
- 이전 시험 대비: {change['change']:+d}점 ({change['direction']})
- 반 평균: {avg}점
- 반 내 순위: {ranking['rank']}등 / {ranking['total']}���

영역별 분석:
{category_text if category_text else '  (영역별 점수 미입력)'}
{prev_text}
{memo_text}"""

    return context


def generate_consultation(student_id: int, exam_id: int) -> str:
    """AI 상담문 초안을 생성한다."""
    context = build_student_context(student_id, exam_id)

    if not context:
        return "학생 데이터를 찾을 수 없습니다."

    client = get_anthropic_client()

    message = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        system="""당신은 국어학원 강사입니다. 학부모에게 전달할 상담문을 작성합니다.

규칙:
- 존댓말 사용 (학부모 대상)
- 전문적이되 따뜻한 어조
- 구조: 1) 전체 성적 변화 요약 → 2) 강점 영역 → 3) 보완 영역 → 4) 구체적 학습 제안
- 모든 언급에 숫자 근거를 포함하세요 (예: "문학 영역이 28점에서 35점으로 7점 상승")
- 이전 상담문이나 메모가 있다면 연속성을 유지하세요
- 300~500자 분량으로 작성하세요
- 막연한 격려보다 구체적 근거 기반으로 작성하세요""",
        messages=[
            {"role": "user", "content": f"다음 학생의 상담문을 작성해주세요.\n\n{context}"}
        ],
    )

    return message.content[0].text
```
